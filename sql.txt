CREATE DATABASE kbolike;

-- create ============

-- 1. 구단 정보 (코드 기반 설계)
CREATE TABLE Team (
    team_code CHAR(2) PRIMARY KEY,
    name VARCHAR(100) NOT NULL UNIQUE,
    CHECK (team_code IN ('KI','SS','LG','HT','NC','KT','LT','DS','WO','HH'))
);

-- 2. 회원 정보 (Passwordless 반영)
CREATE TABLE Users (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    email VARCHAR(100) NOT NULL UNIQUE,
    auth_provider VARCHAR(50) NOT NULL, -- EMAIL, KAKAO, GOOGLE
    name VARCHAR(50) NOT NULL,
    role VARCHAR(20) NOT NULL DEFAULT 'USER'
);

CREATE TABLE Product (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    team_code CHAR(2) NOT NULL,
    name VARCHAR(200) NOT NULL,
    price INT NOT NULL CHECK (price >= 0),
    stock INT NOT NULL CHECK (stock >= 0),
    image_url VARCHAR(500),
    FOREIGN KEY (team_code) REFERENCES Team(team_code)
);

-- 4. 장바구니 (유저당 1개)
CREATE TABLE Cart (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL UNIQUE,
    FOREIGN KEY (user_id) REFERENCES User(id)
);

-- 5. 장바구니 상세
CREATE TABLE CartItem (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    cart_id BIGINT NOT NULL,
    product_id BIGINT NOT NULL,
    quantity INT NOT NULL CHECK (quantity >= 1),
    FOREIGN KEY (cart_id) REFERENCES Cart(id),
    FOREIGN KEY (product_id) REFERENCES Product(id)
);

-- 6. 주문 정보
CREATE TABLE Orders (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    order_num VARCHAR(100) NOT NULL UNIQUE,
    order_date DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(50) NOT NULL, -- READY, PAID, CANCEL
    FOREIGN KEY (user_id) REFERENCES User(id)
);

-- 7. 주문 상세 (주문 시점 가격 보존)
CREATE TABLE OrderItem (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    order_id BIGINT NOT NULL,
    product_id BIGINT NOT NULL,
    order_price INT NOT NULL,
    count INT NOT NULL,
    FOREIGN KEY (order_id) REFERENCES Orders(id),
    FOREIGN KEY (product_id) REFERENCES Product(id)
);

-- 8. 배송 정보
CREATE TABLE Delivery (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    order_id BIGINT NOT NULL UNIQUE,
    receiver_name VARCHAR(50) NOT NULL,
    address VARCHAR(500) NOT NULL,
    tracking_num VARCHAR(100) UNIQUE,
    delivery_status VARCHAR(50) NOT NULL, -- PREPARING, SHIPPING, COMPLETED
    FOREIGN KEY (order_id) REFERENCES Orders(id)
);

-- 9. 결제 정보
CREATE TABLE Payment (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    order_id BIGINT NOT NULL UNIQUE,
    payment_key VARCHAR(255) NOT NULL UNIQUE,
    method VARCHAR(50) NOT NULL,
    amount INT NOT NULL,
    FOREIGN KEY (order_id) REFERENCES Orders(id)
);

-- 10. 게시판
CREATE TABLE Board (
    id BIGINT AUTO_INCREMENT PRIMARY KEY,
    user_id BIGINT NOT NULL,
    title VARCHAR(255) NOT NULL,
    content TEXT NOT NULL,
    created_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES User(id)
);
